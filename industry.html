<!DOCTYPE html>

<html lang="zh-CN">

<head>

    <meta charset="UTF-8">

    <meta name="viewport" content="width=device-width, initial-scale=1.0">

    <title>公司列表</title>

    <link rel="stylesheet" href="css/style.css">

</head>

<body>

    <div class="container">

        <h1 id="industry-title"></h1>

        <div class="company-list" id="company-list">

            </div>

        <a href="index.html" class="back-button">返回主页</a>

    </div>


    <script>

        document.addEventListener('DOMContentLoaded', function() {

            const urlParams = new URLSearchParams(window.location.search);

            const category = urlParams.get('category');

            document.getElementById('industry-title').textContent = `${category} - 公司列表`;


            const companyListDiv = document.getElementById('company-list');

            let companies = [];


            // 这里你可以根据 category 加载不同的公司数据

            if (category === '新能源') {

                companies = [

                    { name: '新能源公司A', id: 'companyA' },

                    { name: '新能源公司B', id: 'companyB' }

                ];

            } else if (category === '消费电子') {

                companies = [

                    { name: '消费电子公司X', id: 'companyX' },

                    { name: '消费电子公司Y', id: 'companyY' }

                ];

            }

            // ... 其他行业分类


            if (companies.length > 0) {

                companies.forEach(company => {

                    const companyLink = document.createElement('a');

                    companyLink.href = `company-detail.html?companyId=${company.id}&companyName=${company.name}`;

                    companyLink.classList.add('company-item');

                    companyLink.textContent = company.name;

                    companyListDiv.appendChild(companyLink);

                });

            } else {

                companyListDiv.innerHTML = '<p>暂无公司数据。</p>';

            }

        });

    </script>

</body>

</html>
